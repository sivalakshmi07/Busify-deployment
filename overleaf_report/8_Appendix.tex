\chapter{APPENDIX}

\section{Minimum Software Requirements}

\begin{itemize}
    \item \textbf{Operating System:} Windows 10/11, macOS, or Linux.
    \item \textbf{Web Browser:} Google Chrome, Mozilla Firefox, Microsoft Edge, or Safari.
    \item \textbf{Runtime Environment:} Node.js (Version 16.x or newer).
    \item \textbf{Database Engine:} MongoDB.
    \item \textbf{Package Manager:} NPM (Node Package Manager).
    \item \textbf{Code Editor:} Visual Studio Code (optional, but recommended for maintenance).
\end{itemize}

\section{Minimum Hardware Requirements}

\begin{itemize}
    \item \textbf{Processor:} Intel Core i3 or equivalent AMD processor (2.0 GHz or higher).
    \item \textbf{Memory (RAM):} 4 GB RAM (8 GB recommended for localized server execution).
    \item \textbf{Storage:} 10 GB of available hard disk space for code repositories, node module dependencies, and temporary database staging.
    \item \textbf{Network:} A stable, high-speed Internet Connection for database fetching and real-time backend updates.
\end{itemize}
